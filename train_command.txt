# node2vec
python train.py --no_cuda \
                --model node2vec \
                --batch_size 256 \
                --epochs 1000 \
                --lr 0.001 \
                --weight_decay 0.0005 \
                --output 256 \
                --walk_length 50 \
                --context_size 10 \
                --walks_per_node 10 \
                --is_parallel \
                --data ./data/wordnet/edge/synsets \
                --dictionary ./data/wordnet/dictionary/synsets.txt \
                1>node2vec_log.txt 2>&1 &
